class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null) return l2;
        else if (l2 == null) return l1;
        else if(l1.val<l2.val){
            l1.next = mergeTwoLists(l1.next,l2);
            return l1;
        }else{
            l2.next = mergeTwoLists(l1,l2.next);

        }
    }

class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
         ListNode node = new ListNode();
        ListNode p = node;
        while (l1!=null || l2!=null){
            if (l1 == null){
                node.next = l2;
                return p.next;
            }
            else if(l2 == null){
                node.next = l1;
                return p.next;
            }
            if(l1.val<l2.val) {
                node.next = l1;
                node = node.next;
                l1 = l1.next;
            }else{
                node.next = l2;
                node = node.next;
                l2 = l2.next;
            }
        }
        return p.next;
    }
}
